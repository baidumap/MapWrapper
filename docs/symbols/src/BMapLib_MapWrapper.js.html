1 /**
  2  * @fileoverview MapWrapper类提供了将Google或GPS坐标形式的Marker添加到地图上的功能。
  3  * 用户可以直接通过该类提供的addOverlay方法,添加Google或GPS坐标形式的Marker到地图上,
  4  * 不再需要先将Google或GPS坐标转化为百度坐标，然后再添加的操作。
  5  * 主入口类是<a href="symbols/BMapLib.MapWrapper.html">MapWrapper</a>，
  6  * 基于Baidu Map API 1.2。
  7  *
  8  * @author Baidu Map Api Group 
  9  * @version 1.2
 10  */
 11 
 12 /** 
 13  * @namespace BMap的所有library类均放在BMapLib命名空间下
 14  */
 15 var BMapLib = window.BMapLib = BMapLib || {};
 16 
 17 (function() { 
 18 
 19     /**
 20      * GPS坐标系类型
 21      */
 22     BMapLib.COORD_TYPE_GPS  = 0;
 23     
 24     /**
 25      * Google 坐标系类型
 26      */
 27     BMapLib.COORD_TYPE_GOOGLE = 2;
 28 
 29     /**
 30      * Baidu 坐标系类型
 31      */
 32     var COORD_TYPE_BAIDU = 4;
 33     
 34     /**
 35      * 每次最大可以转化的坐标点数量
 36      */
 37     var CONVERT_NUM_ONCE = 20;
 38     
 39     
 40     /** 
 41      * @exports MapWrapper as BMapLib.MapWrapper 
 42      */
 43     var MapWrapper =    
 44 
 45     /**
 46      * MapWrapper类的构造函数
 47      * @class 将Google或GPS坐标形式的Marker添加到地图上,
 48      * 实例化该类后，即可调用该类提供的addOverlay方法添加GPS或Google坐标形式的Marker。
 49      * 
 50      * @constructor
 51      * @param {Map} map BMap.Map的实例对象
 52      * @param {Number} coordType 需要转化的坐标类型，如：GPS坐标类型 | Google坐标类型
 53      *     
 54      */
 55     BMapLib.MapWrapper = function(map, coordType){
 56         this._map = map;
 57         this._coordType = coordType;
 58 
 59         //临时对象，转化前的mkr都挂载到此对象上，转化完毕后再删除掉
 60         this._hashObject = {};
 61 
 62         //缓存数组，用来累计每次20个坐标点，然后一同发送请求
 63         this._arrCache = [];
 64     }
 65 
 66     /*
 67      * 计数器，用来唯一标识每个发送的坐标点序列号
 68      */
 69     MapWrapper.prototype._uiqueNum = 0;
 70 
 71     /**
 72      * 是否可以发送请求,用来处理累计发送请求的标识
 73      */
 74     MapWrapper.prototype._canSendRequest = true;
 75 
 76     /**
 77      * 将覆盖物添加到地图上。此函数首先将其他坐标转化为百度坐标，然后再将覆盖物添加到地图上。
 78      * @param {Overlay} overlay 覆盖物对象，目前只支持Marker类对象 
 79      * @return 无返回值
 80      */
 81     MapWrapper.prototype.addOverlay = function(overlay){
 82         //目前仅支持Marker类型
 83         if(!overlay || !(overlay instanceof BMap.Marker)){
 84             return;
 85         }
 86 
 87         var data = {
 88             'overlay'    : overlay,        
 89             'uiqueNum'   : this._uiqueNum //唯一标识该坐标点
 90         };
 91         
 92         this._arrCache.push(data);
 93         this._hashObject['guid' + this._uiqueNum] = overlay; //记录在hash对象中，回调后使用
 94 
 95         this._uiqueNum++;
 96         
 97         var me = this;    
 98         if(me._canSendRequest){
 99             me._canSendRequest = false;
100 
101             window.setTimeout(function(){
102                 me._canSendRequest = true;
103                 me._delayRequest();            
104             }, 50);
105         }
106     }
107 
108     /**
109      * 延迟发送请求
110      */
111     MapWrapper.prototype._delayRequest = function(){
112         var arrCache = this._arrCache;
113         var group = Math.ceil(arrCache.length / CONVERT_NUM_ONCE); //分组个数            
114         
115         for(var i = 0; i < group; i++){
116             var arrX = [], arrY = [], arrGuid = [];
117             
118             var minIndex = i * CONVERT_NUM_ONCE;
119             var maxIndex = -1;
120             if(arrCache.length - (i + 1) * CONVERT_NUM_ONCE > 0){
121                 maxIndex = (i + 1) * CONVERT_NUM_ONCE;
122             } else {
123                 maxIndex = arrCache.length;
124             }
125             
126             for(var j = minIndex; j < maxIndex; j++){            
127                 var data = arrCache[j];            
128                 var x = data.overlay.getPosition().lng;
129                 var y = data.overlay.getPosition().lat;
130                 
131                 arrX.push(x);
132                 arrY.push(y);            
133                 arrGuid.push('guid' + data.uiqueNum);
134             }
135 
136             var strX = arrX.join(",");
137             var strY = arrY.join(",");
138             
139             //坐标点参数
140             var jsonParam = {
141                 x:      strX,
142                 y:      strY,
143                 from:   this._coordType,
144                 to:     COORD_TYPE_BAIDU,
145                 mode:   1
146             };
147             
148             //用户自定义参数
149             var userData = {
150                 'guids': arrGuid 
151             };
152             
153             var me = this;
154             MapWrapper._SearchRequestMgr.request(function(json, userData){
155                 me._requestCbk(json, userData);
156             }, jsonParam, userData);        
157         }    
158 
159         this._arrCache.length = 0; //将数组归0
160     }
161 
162     /**
163      * 坐标转化后的回调函数
164      * @param {Json} json 转化后的json数据
165      * @param {Json} userData 请求发送前一些参数，在回调函数中使用
166      * 
167      */
168     MapWrapper.prototype._requestCbk = function(json, userData){
169         if(json && json instanceof Array){
170             var guids = userData.guids;
171             for(var i = 0; i < json.length; i++){
172                 var item = json[i];
173                 if(item && item.error == 0){
174                     var newX = item.x, newY = item.y;
175                     var newPt = new BMap.Point(newX, newY);
176                     
177                     var guid = guids[i];
178                     if(this._hashObject[guid]){
179                         var mkr = this._hashObject[guid];
180                         mkr.setPosition(newPt);
181 
182                         this._map.addOverlay(mkr);
183 
184                         delete this._hashObject[guid];
185                     }
186 
187                 }
188             }
189         }
190     }
191 
192     /**
193      * json对象转化为url字串
194      * @param {Json} json 需要转化的json对象
195      * @param {Function} encode 编码函数
196      */
197     MapWrapper._jsonToQuery = function(json, encode){
198       var s = [];
199       encode = encode || function(v){return v};
200       for (var n in json){
201         s.push(n + "=" + encode(json[n]));
202       }
203       return s.join("&");
204     }
205 
206     /**
207      * 请求管理模块 
208      */
209     MapWrapper._SearchRequestMgr = {
210       /**
211        * 请求地址
212        */
213       COORD_CONVERT_URL: "http://api.map.baidu.com/ag/coord/convert",  
214       /**
215        * 请求函数
216        * @param {Function} cbk 回调函数
217        * @param {Object} params 附加请求参数, 在回调调用时传递给回调函数
218        * @param {Object} userData 用户自定义数据, 在回调调用时传递给回调函数
219        * @param {String} path 附加请求路径，可选
220        */
221       request: function(cbk, params, userData, path){    
222         var timeStamp = (Math.random() * 100000).toFixed(0);// 生成随机数
223         
224         MapWrapper._cbkMount["_cbk" + timeStamp] = function(json){// 全局回调函数
225           userData = userData || {};
226           cbk && cbk(json, userData);
227           delete MapWrapper._cbkMount["_cbk" + timeStamp];
228         };
229 
230         path = path || "";    
231         
232         var reqParam = MapWrapper._jsonToQuery(params, encodeURIComponent);    
233 
234         var me = this,
235             url = me.COORD_CONVERT_URL + path + "?" + reqParam + 
236                         "&ie=utf-8&oue=1&res=api&callback=BMapLib.MapWrapper._cbkMount._cbk" + timeStamp;
237             
238         me.createScript(url);
239       },
240       /**
241        * 动态创建script标签，发送请求
242        * @param {String} url script标签对应的url
243        */
244       createScript: function(url) {    
245         var script = document.createElement("script");
246         script.src = url;
247         script.setAttribute('type', 'text/javascript');
248         script.setAttribute('charset', 'utf-8');
249 
250         // 脚本加载完成后进行移除
251         if (script.addEventListener) {
252           script.addEventListener('load', function(e) {
253             var t = e.target;
254             t.parentNode.removeChild(t);
255           }, false);
256         }
257         else if (script.attachEvent) {
258           script.attachEvent('onreadystatechange', function(e) {
259             var t = window.event.srcElement;
260             if (t && (t.readyState == 'loaded' || t.readyState == 'complete')) {
261               t.parentNode.removeChild(t);
262             }
263           });
264         }
265         // 使用setTimeout解决ie6无法发送问题
266         setTimeout(function() {        
267           document.getElementsByTagName('head')[0].appendChild(script);      
268           script = null;
269         }, 1);
270       }
271     };
272 
273     /*
274      * 全局回调函数挂载点
275      */
276     MapWrapper._cbkMount = {};
277 
278 })();//闭包结束
